# validationplot 함수를 사용하여 cross-validation score를 그래프로 나타냅니다.
# val.type = "MSEP" 를 하면 CV MSE가 그려질 것
# "RMSEP" 를 하면 MSEP의 제곱근 값을 "R2"를 하면 R^2 값을 그릴 것

validationplot(pcr.fit, val.type = "MSEP")
points(16, 347.6^2, pch=1, cex = 3, col = 4)
text(17, 355^2, "347.6^2")
points(1, 348.9^2, pch=4, cex=3, col=5)
text(2, 344^2, "348.9^2")

M = 16일 때 cv error가 가장 작은 것을 볼 수 있다. 전체 p가 19임을 고려할 때, 16개의 예측 변수는 PCR을 적용하기에는 개수가 많은 편이다. 그러나 M = 1 일 때에도 cv error가 작은 것을 보아, 적은 수의 변수를 사용하는 모델도 충분할 것으로 예상해 볼 수 있다.

variance가 설명되는 비율은 M의 크기에 따라 어떻게 변하는가? summary 함수를 통해 살펴보면, M = 1 일 때 예측 변수의 분산이 38.31% 정도 밖에 설명이 되지 않으나, M = 6일 때는 88.63%로 설명되는 비율이 증가한다. M = 19일 때는 당연히 100% 겠지?

set.seed(1)
x = model.matrix(Salary~., Hitters)[,-1]
y = Hitters$Salary
train = sample(1:nrow(x), nrow(x)/2)
test = (-train)
y.test = y[test]

# Training data로 모델 생성
pcr.fit = pcr(Salary~., data=Hitters, subset=train, scale=TRUE, validation="CV")
validationplot(pcr.fit, val.type="MSEP")
points(7, 381.3^2, pch=1, cex=3, col=4)
text(7, 150000, "381.3^2")

# Test-MSE를 계산한다
# Ridge regression: 96010, Lasso regression: 100743
pcr.pred = predict(pcr.fit, x[test,], ncomp=7)
mean((pcr.pred - y.test)^2)
96556.2191395342
